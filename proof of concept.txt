const Card = ({ show, isOwned, onAction, actionType = 'add', compact = false }) => {
  const [isFlashing, setIsFlashing] = useState(false);

  const handleClick = (e) => {
    if (!isOwned && actionType === 'add') {
      setIsFlashing(true);
      setTimeout(() => setIsFlashing(false), 400);
    }
    if (onAction) onAction(show.id);
  };

  return (
    <div 
      className={`relative group cursor-pointer w-full flex-shrink-0 snap-center ${compact ? 'aspect-square' : 'aspect-[2/3] w-[130px] md:w-[150px]'} rounded-[20px] z-10 transition-transform duration-150 active:scale-95`}
      onClick={handleClick}
    >
      <div className={`absolute inset-0 rounded-[20px] ring-2 ring-inset ring-white/10 overflow-hidden bg-[#1a1a24] flex flex-col shadow-lg ${isFlashing ? 'animate-pulse bg-white' : ''}`}>
        
        {/* Shiny Effect Overlay */}
        <div className="absolute inset-0 opacity-0 group-hover:opacity-100 transition-opacity duration-500 pointer-events-none z-20 overflow-hidden rounded-[20px]">
          <div className="absolute top-0 -left-[100%] w-1/2 h-full bg-gradient-to-r from-transparent via-white/20 to-transparent skew-x-[-20deg] group-hover:animate-shine" />
        </div>

        {/* Image */}
        <div className="relative h-full w-full overflow-hidden bg-black">
          <img src={show.image} alt={show.title} className="w-full h-full object-cover" />
          
          {/* Overlay Button */}
          {!compact && actionType === 'add' && (
            <button 
              className={`absolute top-2 right-2 z-30 flex-shrink-0 w-8 h-8 rounded-lg font-black transition-all duration-200 flex items-center justify-center
                ${isOwned 
                  ? 'bg-black/60 text-white cursor-default backdrop-blur-md border border-white/10' 
                  : 'bg-[#4ade80] text-black hover:bg-[#22c55e] hover:scale-105 active:scale-95 shadow-lg'

              onClick={(e) => { e.stopPropagation(); handleClick(e); }}
            >
              {isOwned ? <Check size={16} strokeWidth={3} /> : <Plus size={20} strokeWidth={4} />}

      </div>

const RankGame = ({ unsortedShows, totalUnsorted, onSort, onFinish }) => {
  const currentShow = unsortedShows[0];
  const [animatingOut, setAnimatingOut] = useState(false);

  if (!currentShow) return null;

  const handleTierClick = (tier) => {
    if (animatingOut) return;
    setAnimatingOut(true);
    setTimeout(() => {
      onSort(currentShow.id, tier);
      setAnimatingOut(false);
    }, 300);
  };

  const progressPercent = ((totalUnsorted - unsortedShows.length) / totalUnsorted) * 100;

  return (
    <div className="fixed inset-0 z-50 bg-black/95 backdrop-blur-xl flex flex-col items-center justify-center p-6">
      <div className="absolute top-8 right-8 cursor-pointer p-3 bg-white/10 rounded-full hover:bg-white/20 active:scale-90 transition-transform" onClick={onFinish}>
        <X size={24} className="text-white" />
      </div>
      
      {/* Progress Bar */}
      <div className="absolute top-12 left-1/2 -translate-x-1/2 w-[60%] max-w-[200px]">
        <div className="flex justify-between items-end mb-2">
          <span className="font-black uppercase tracking-widest text-xs text-gray-400">Ranking Queue</span>
          <span className="font-black text-sm text-[#4ade80]">{totalUnsorted - unsortedShows.length}/{totalUnsorted}</span>
        </div>
        <div className="h-2 w-full bg-white/10 rounded-full overflow-hidden">
          <div className="h-full bg-[#4ade80] transition-all duration-300" style={{ width: `${progressPercent}%` }} />
        </div>
      </div>

      <div className={`w-full max-w-[260px] mt-12 transition-all duration-300 ${animatingOut ? 'scale-50 opacity-0 -translate-y-20' : 'scale-100 opacity-100'}`}>
        <Card show={currentShow} actionType="none" />
      </div>

      <h2 className="mt-8 text-xl font-black uppercase tracking-widest text-white/50 text-center">Where does it rank?</h2>

      <div className="mt-6 flex flex-wrap justify-center gap-3 w-full max-w-sm">
        {Object.keys(TIER_COLORS).map(tier => (
          <button
            key={tier}
            onClick={() => handleTierClick(tier)}
            className={`w-[72px] h-[72px] rounded-2xl font-black text-4xl flex items-center justify-center transition-transform active:scale-90 shadow-xl ${TIER_COLORS[tier]}`}
          >
            {tier}
          </button>
        ))}
      </div>
    </div>
  );
};

export default function App() {
  const [activeTab, setActiveTab] = useState('discover');
  const [ownedIds, setOwnedIds] = useState([]);
  const [tierData, setTierData] = useState({ S: [], A: [], B: [], C: [], D: [] });
  const [top8, setTop8] = useState(Array(8).fill(null));
  
  // Profile State
  const [profile, setProfile] = useState({ name: 'Player One', colorIndex: 0 });
  const [isEditingProfile, setIsEditingProfile] = useState(false);
  const [searchQuery, setSearchQuery] = useState('');
  
  // Collection State
  const [collectionSearchQuery, setCollectionSearchQuery] = useState('');
  const [collectionFilter, setCollectionFilter] = useState('All');

  // Modal States
  const [isRanking, setIsRanking] = useState(false);
  const [selectingForSlot, setSelectingForSlot] = useState(null); // index 0-7
  const [initialUnsortedCount, setInitialUnsortedCount] = useState(0);

  // Derived Data
  const ownedShows = useMemo(() => MOCK_SHOWS.filter(s => ownedIds.includes(s.id)), [ownedIds]);
  
  // Extract unique genres for the filter
  const GENRES = useMemo(() => {
    const genres = new Set(ownedShows.map(s => s.genre));
    return Array.from(genres).sort();
  }, [ownedShows]);

  const unsortedShows = useMemo(() => {
    const sortedIds = Object.values(tierData).flat();
    return ownedShows.filter(s => !sortedIds.includes(s.id));
  }, [ownedShows, tierData]);

  const filteredShows = useMemo(() => {
    if (!searchQuery) return null;
    return MOCK_SHOWS.filter(s => s.title.toLowerCase().includes(searchQuery.toLowerCase()));
  }, [searchQuery]);

  const filteredCollection = useMemo(() => {
    return ownedShows.filter(show => {
      const matchesSearch = show.title.toLowerCase().includes(collectionSearchQuery.toLowerCase());
      const matchesFilter = collectionFilter === 'All' || show.genre === collectionFilter;
      return matchesSearch && matchesFilter;
    });
  }, [ownedShows, collectionSearchQuery, collectionFilter]);

  // Rankings Stats Data
  const totalRanked = Object.values(tierData).flat().length;
  const sTierCount = tierData.S.length;
  const topHeavyPercent = totalRanked > 0 ? Math.round(((tierData.S.length + tierData.A.length) / totalRanked) * 100) : 0;

  // Actions
  const handleAddShow = (id) => {
    if (!ownedIds.includes(id)) {
      setOwnedIds([...ownedIds, id]);
    }
  };

  const startRanking = () => {
    setInitialUnsortedCount(unsortedShows.length);
    setIsRanking(true);
  };

  const handleRankSubmit = (showId, tier) => {
    setTierData(prev => {
      const newData = { ...prev };
      Object.keys(newData).forEach(k => {
        newData[k] = newData[k].filter(id => id !== showId);
      });
      newData[tier] = [...newData[tier], showId];
      return newData;
    });
  };

  const handleRemoveFromTier = (showId, currentTier) => {
    setTierData(prev => ({
      ...prev,
      [currentTier]: prev[currentTier].filter(id => id !== showId)
    }));
  };

  const handleAssignTop8 = (showId) => {
    if (selectingForSlot !== null) {
      setTop8(prev => {
        const newTop8 = [...prev];
        const existingIndex = newTop8.indexOf(showId);
        if(existingIndex > -1) newTop8[existingIndex] = null; 
        newTop8[selectingForSlot] = showId;
        return newTop8;
      });
      setSelectingForSlot(null);
    }
  };

  const handleRemoveTop8 = (index) => {
    setTop8(prev => {
      const newTop8 = [...prev];
      newTop8[index] = null;
      return newTop8;
    });
  };

  // Inject Custom Styles for animations
  useEffect(() => {
    const style = document.createElement('style');
    style.innerHTML = `
      @keyframes shine {
        100% { left: 200%; }
      }
      .animate-shine {
        animation: shine 1.5s ease-in-out infinite;
      }
      .no-scrollbar::-webkit-scrollbar { display: none; }
      .no-scrollbar { -ms-overflow-style: none; scrollbar-width: none; }
    `;
    document.head.appendChild(style);
    return () => document.head.removeChild(style);
  }, []);

  return (
    <div className="min-h-screen bg-[#0a0a0c] text-white font-sans selection:bg-[#4ade80] flex justify-center w-full">
      {/* Mobile-sized App Container */}
      <div className="w-full max-w-md relative pb-28 shadow-2xl bg-[#0f0f13] min-h-screen border-x border-white/5 overflow-x-hidden">
        
        {/* --- DISCOVER VIEW --- */}
        {activeTab === 'discover' && (
          <div className="animate-in fade-in duration-300 flex flex-col min-h-screen">
            <Header title="Discover" showSearch={true} onSearch={setSearchQuery} />
            
            <div className="pt-4 flex-1">
              {searchQuery && filteredShows ? (
                <div className="px-4">
                   <h2 className="font-black tracking-widest text-sm uppercase text-gray-400 mb-4">Search Results ({filteredShows.length})</h2>
                   <div className="grid grid-cols-2 gap-4 pb-8">
                     {filteredShows.map(show => (
                        <Card key={show.id} show={show} isOwned={ownedIds.includes(show.id)} onAction={handleAddShow} />
                     ))}
                   </div>
                </div>
              ) : (
                <>
                  <CarouselRow title="Recommended for You" shows={MOCK_SHOWS.filter(s => s.category === 'recommended')} ownedIds={ownedIds} handleAction={handleAddShow} useLandscape={true} />
                  <CarouselRow title="Trending Now" shows={MOCK_SHOWS.filter(s => s.category === 'trending')} ownedIds={ownedIds} handleAction={handleAddShow} />
                  <CarouselRow title="Animation" shows={MOCK_SHOWS.filter(s => s.category === 'animation')} ownedIds={ownedIds} handleAction={handleAddShow} />
                  <CarouselRow title="Classics" shows={MOCK_SHOWS.filter(s => s.category === 'classics')} ownedIds={ownedIds} handleAction={handleAddShow} />
                </>
              )}
            </div>
          </div>
        )}

        {/* --- COLLECTION VIEW (Full Inventory) --- */}
        {activeTab === 'collection' && (
          <div className="animate-in fade-in duration-300 flex flex-col min-h-screen">
            <Header title="Collection" showSearch={true} onSearch={setCollectionSearchQuery} />
            
            <div className="p-4 flex-1">
              {/* Filters */}
              <div className="flex items-center gap-2 mb-6 overflow-x-auto no-scrollbar pb-2">
                <div className="flex items-center justify-center p-2 bg-white/5 rounded-xl border border-white/10 mr-1 flex-shrink-0">
                  <Filter size={18} className="text-gray-400" />
                </div>
                <button 
                  onClick={() => setCollectionFilter('All')}
                  className={`px-4 py-2 rounded-xl font-black text-sm uppercase tracking-widest whitespace-nowrap transition-colors border-2
                    ${collectionFilter === 'All' ? 'bg-white text-black border-white' : 'bg-transparent text-gray-400 border-white/10 hover:border-white/30'}`}
                >
                  All Genres
                </button>
                {GENRES.map(genre => (
                  <button 
                    key={genre}
                    onClick={() => setCollectionFilter(genre)}
                    className={`px-4 py-2 rounded-xl font-black text-sm uppercase tracking-widest whitespace-nowrap transition-colors border-2
                      ${collectionFilter === genre ? 'bg-[#4ade80] text-black border-[#4ade80]' : 'bg-transparent text-gray-400 border-white/10 hover:border-white/30'}`}
                  >
                    {genre}
                  </button>
                ))}
              </div>
              
              {/* Grid */}
              {ownedShows.length === 0 ? (
                <div className="flex flex-col items-center justify-center py-20 opacity-50 bg-[#1a1a24] rounded-3xl border border-white/5">
                  <Package size={64} className="mb-4" />
                  <p className="font-black tracking-widest text-lg uppercase text-white">Collection Empty</p>
                  <p className="font-bold text-xs uppercase tracking-widest text-gray-400 mt-2">Go discover some shows</p>
                </div>
              ) : filteredCollection.length === 0 ? (
                <div className="flex flex-col items-center justify-center py-20 opacity-50">
                  <Search size={48} className="mb-4 text-gray-500" />
                  <p className="font-black tracking-widest text-sm uppercase">No matches found</p>
                </div>
              ) : (
                <div className="grid grid-cols-2 gap-4 pb-8">
                   {filteredCollection.map(show => (
                      <Card key={show.id} show={show} isOwned={true} actionType="none" />
                   ))}
                </div>
              )}
            </div>
          </div>
        )}

        {/* --- PROFILE VIEW (Identity + Showcase) --- */}
        {activeTab === 'profile' && (
          <div className="animate-in fade-in duration-300">
            <Header title="My Profile" />
            
            <div className="p-4">
              {/* Player Identity Area */}
              <div className="relative mb-8 bg-[#1a1a24] p-5 rounded-[24px] border border-white/10 shadow-xl overflow-hidden">
                <div className={`absolute top-0 right-0 w-32 h-32 bg-gradient-to-br ${PROFILE_COLORS[profile.colorIndex]} opacity-20 blur-3xl rounded-full translate-x-10 -translate-y-10`}></div>
                
                <div className="flex items-center gap-4 relative z-10">
                  <div className={`w-20 h-20 rounded-[20px] bg-gradient-to-tr ${PROFILE_COLORS[profile.colorIndex]} flex items-center justify-center shadow-lg border-2 border-white/20`}>
                    <User size={40} className="text-white drop-shadow-md" />
                  </div>
                  
                  <div className="flex-1">
                    {isEditingProfile ? (
                      <input 
                        type="text" 
                        value={profile.name} 
                        onChange={(e) => setProfile({...profile, name: e.target.value.toUpperCase()})}
                        className="w-full bg-black/50 border border-white/20 rounded-lg px-3 py-1 font-black uppercase tracking-tighter text-2xl text-white outline-none focus:border-[#4ade80] mb-2"
                        maxLength={12}
                      />
                    ) : (
                      <h1 className="text-3xl font-black uppercase tracking-tighter text-white leading-none mb-1">{profile.name}</h1>
                    )}
                    
                    <div className="flex items-center gap-3">
                      <span className="text-[10px] bg-white/10 px-2 py-0.5 rounded font-black text-gray-300 tracking-widest uppercase">Level {Math.floor(ownedIds.length / 3) + 1}</span>
                      <span className="text-[10px] font-bold text-[#4ade80] tracking-widest uppercase">{ownedIds.length}/{MOCK_SHOWS.length} Added</span>
                    </div>
                  </div>

                  <button 
                    onClick={() => setIsEditingProfile(!isEditingProfile)}
                    className={`p-3 rounded-full transition-colors ${isEditingProfile ? 'bg-[#4ade80] text-black' : 'bg-white/10 text-white hover:bg-white/20'}`}
                  >
                    {isEditingProfile ? <Check size={20} /> : <Edit3 size={20} />}
                  </button>
                </div>

                {isEditingProfile && (
                  <div className="mt-6 pt-4 border-t border-white/10 animate-in slide-in-from-top-2 duration-200">
                    <div className="flex items-center gap-2 mb-3">
                      <Palette size={14} className="text-gray-400" />
                      <span className="text-xs font-bold text-gray-400 uppercase tracking-widest">Banner Color</span>
                    </div>
                    <div className="flex gap-3 overflow-x-auto no-scrollbar pb-2">
                      {PROFILE_COLORS.map((color, idx) => (
                        <button 
                          key={idx}
                          onClick={() => setProfile({...profile, colorIndex: idx})}
                          className={`w-12 h-12 rounded-xl flex-shrink-0 bg-gradient-to-tr ${color} border-2 transition-transform active:scale-90
                            ${profile.colorIndex === idx ? 'border-white scale-110' : 'border-transparent opacity-50'}
                          `}
                        />
                      ))}
                    </div>
                  </div>
                )}
              </div>

              {/* Showcase (Top 8) */}
              <div className="mb-8">
                <div className="flex items-center justify-between mb-4 px-1">
                  <div className="flex items-center gap-2">
                    <Star size={18} className="text-yellow-400 fill-yellow-400" />
                    <h2 className="font-black tracking-widest text-lg uppercase text-white">Loadout</h2>
                  </div>
                  <span className="text-xs font-bold text-gray-500 uppercase tracking-widest">Top 8</span>
                </div>
                
                <div className="grid grid-cols-4 gap-3 pb-8">
                  {top8.map((showId, index) => {
                    const show = showId ? MOCK_SHOWS.find(s => s.id === showId) : null;
                    return (
                      <div 
                        key={index} 
                        className={`aspect-square rounded-[16px] relative cursor-pointer overflow-hidden transition-transform active:scale-95 group
                          ${!show ? 'border-2 border-dashed border-white/20 bg-[#1a1a24] hover:bg-white/10 flex items-center justify-center' : ''}
                        `}
                        onClick={() => show ? handleRemoveTop8(index) : setSelectingForSlot(index)}
                      >
                        {!show ? (
                          <Plus size={24} className="text-white/30 group-hover:text-white/60 transition-colors" />
                        ) : (
                          <>
                            <Card show={show} compact={true} actionType="none" />
                            <div className="absolute inset-0 bg-red-500/80 opacity-0 group-hover:opacity-100 transition-opacity flex items-center justify-center backdrop-blur-sm">
                              <X size={24} className="text-white" />
                            </div>
                          </>
                        )}
                      </div>
                    )
                  })}
                </div>
              </div>
            </div>
          </div>
        )}

        {/* --- RANKINGS VIEW --- */}
        {activeTab === 'rankings' && (
          <div className="animate-in fade-in duration-300 min-h-screen flex flex-col">
            <Header title="Rankings" />
            
            <div className="p-4 flex flex-col">
              {/* Action Banner */}
              {unsortedShows.length > 0 ? (
                <div className="mb-8 bg-gradient-to-r from-[#4ade80] to-emerald-600 rounded-[24px] p-1 shadow-[0_0_20px_rgba(74,222,128,0.2)] animate-pulse">
                  <div className="bg-[#0f0f13]/90 backdrop-blur-md rounded-[20px] p-4 flex items-center justify-between">
                    <div>
                      <h3 className="font-black text-white uppercase tracking-widest text-sm">Unranked Shows</h3>
                      <p className="text-emerald-300 text-xs font-bold mt-1 uppercase tracking-widest">{unsortedShows.length} waiting to be ranked</p>
                    </div>
                    <button 
                      onClick={startRanking}
                      className="bg-white text-black px-4 py-2.5 rounded-xl font-black text-sm uppercase tracking-widest hover:scale-105 transition-transform active:scale-95 flex items-center gap-2 shadow-lg"
                    >
                      <PlaySquare size={16} className="fill-black" /> Rank Now
                    </button>
                  </div>
                </div>
              ) : ownedIds.length === 0 ? (
                <div className="mb-8 bg-[#1a1a24] rounded-[24px] p-1 border border-white/10">
                  <div className="bg-[#0f0f13] rounded-[20px] p-4 flex items-center justify-between">
                    <div>
                      <h3 className="font-black text-white uppercase tracking-widest text-sm">Nothing to Rank</h3>
                      <p className="text-gray-500 text-xs font-bold mt-1 uppercase tracking-widest">Go find some shows!</p>
                    </div>
                    <button 
                      onClick={() => setActiveTab('discover')}
                      className="bg-white/10 text-white px-4 py-2.5 rounded-xl font-black text-sm uppercase tracking-widest hover:bg-white/20 transition-colors flex items-center gap-2"
                    >
                      <Compass size={16} /> Discover
                    </button>
                  </div>
                </div>
              ) : (
                <div className="mb-8 bg-[#1a1a24] rounded-[24px] p-1 border border-white/10">
                  <div className="bg-[#0f0f13] rounded-[20px] p-4 flex items-center justify-between">
                    <div>
                      <h3 className="font-black text-white uppercase tracking-widest text-sm">All Caught Up!</h3>
                      <p className="text-[#4ade80] text-xs font-bold mt-1 uppercase tracking-widest">0 Unranked Shows</p>
                    </div>
                    <div className="w-10 h-10 rounded-full bg-[#4ade80]/20 flex items-center justify-center mr-1">
                      <Check size={20} className="text-[#4ade80]" strokeWidth={3} />
                    </div>
                  </div>
                </div>
              )}

              {/* Rankings Dashboard Stats */}
              <div className="grid grid-cols-2 gap-3 mb-8">
                <div className="bg-[#1a1a24] p-4 rounded-[20px] border border-white/5">
                  <span className="text-[10px] font-bold text-gray-500 uppercase tracking-widest">Shows Ranked</span>
                  <p className="text-3xl font-black mt-1 text-white">{totalRanked}</p>
                </div>
                <div className="bg-[#1a1a24] p-4 rounded-[20px] border border-white/5">
                  <span className="text-[10px] font-bold text-gray-500 uppercase tracking-widest">Top Heavy</span>
                  <p className="text-3xl font-black mt-1 text-transparent bg-clip-text bg-gradient-to-r from-[#4ade80] to-emerald-400">{topHeavyPercent}%</p>
                </div>
                <div className="col-span-2 bg-gradient-to-r from-rose-500/20 to-transparent p-4 rounded-[20px] border border-rose-500/20 flex items-center justify-between">
                  <div>
                     <span className="text-[10px] font-bold text-rose-300 uppercase tracking-widest">S-Tier Shows</span>
                     <p className="text-xl font-black mt-1 text-white">{sTierCount} Masterpieces</p>
                  </div>
                  <button className="w-10 h-10 rounded-full bg-white/10 flex items-center justify-center hover:bg-white/20 active:scale-90 transition-all">
                    <Share size={18} className="text-white" />
                  </button>
                </div>
              </div>

              {/* The Tier List */}
              <div className="flex flex-col gap-4 pb-12">
                {Object.entries(tierData).map(([tier, ids]) => (
                  <div key={tier} className="flex bg-[#1a1a24] rounded-2xl overflow-hidden border border-white/5 min-h-[100px] shadow-lg">
                    {/* Tier Label */}
                    <div className={`w-16 flex-shrink-0 flex items-center justify-center font-black text-4xl shadow-xl z-10 border-r border-black/20 ${TIER_COLORS[tier]}`}>
                      {tier}
                    </div>
                    {/* Tier Items */}
                    <div className="flex-1 flex items-center gap-3 p-3 overflow-x-auto no-scrollbar relative shadow-[inset_10px_0_20px_rgba(0,0,0,0.5)] bg-black/20">
                       {ids.length === 0 ? (
                         <span className="text-white/10 font-black uppercase tracking-widest text-sm m-auto">Empty</span>
                       ) : (
                         ids.map(id => {
                           const show = MOCK_SHOWS.find(s => s.id === id);
                           return (
                             <div key={id} className="w-[70px] flex-shrink-0 relative group cursor-pointer" onClick={() => handleRemoveFromTier(id, tier)}>
                               <Card show={show} compact={true} actionType="none" />
                               {/* Hover Remove Overlay */}
                               <div className="absolute inset-0 bg-red-500/90 opacity-0 group-hover:opacity-100 transition-opacity rounded-[20px] flex items-center justify-center backdrop-blur-sm">
                                 <X size={24} className="text-white" />
                               </div>
                             </div>
                           )
                         })
                       )}
                    </div>
                  </div>
                ))}
              </div>
            </div>
          </div>
        )}

        {/* --- OVERLAYS & MODALS --- */}
        {isRanking && (
          <RankGame 
            unsortedShows={unsortedShows} 
            totalUnsorted={initialUnsortedCount}
            onSort={handleRankSubmit} 
            onFinish={() => setIsRanking(false)} 
          />
        )}

        {selectingForSlot !== null && (
          <SelectionModal 
            shows={ownedShows} 
            onSelect={handleAssignTop8} 
            onClose={() => setSelectingForSlot(null)} 
            title="SELECT FOR LOADOUT"
          />
        )}

        {/* --- BOTTOM NAVIGATION --- */}
        <div className="fixed bottom-6 left-1/2 -translate-x-1/2 w-[90%] max-w-[360px] bg-[#1a1a24]/95 backdrop-blur-xl border border-white/10 rounded-full p-2 flex justify-between items-center shadow-[0_20px_40px_rgba(0,0,0,0.8)] z-40">
          {[
            { id: 'discover', icon: Compass, label: 'Discover' },
            { id: 'collection', icon: Package, label: 'Collection', badge: ownedIds.length },
            { id: 'rankings', icon: Trophy, label: 'Rankings', badge: unsortedShows.length > 0 ? '!' : null },
            { id: 'profile', icon: User, label: 'Profile' }
          ].map(tab => {
            const isActive = activeTab === tab.id;
            const Icon = tab.icon;
            return (
              <button
                key={tab.id}
                onClick={() => setActiveTab(tab.id)}
                className={`relative p-3 rounded-full flex flex-col items-center justify-center transition-all duration-300 flex-1 h-14
                  ${isActive ? 'bg-white/10' : 'hover:bg-white/5'}
                `}
              >
                <Icon 
                  size={isActive ? 24 : 20} 
                  className={`transition-colors duration-300 ${isActive ? 'text-[#4ade80] drop-shadow-[0_0_8px_rgba(74,222,128,0.8)]' : 'text-gray-400'}`} 
                  strokeWidth={isActive ? 2.5 : 2}
                />
                
                {/* Notification Badge */}
                {tab.badge && !isActive && (
                   <div className="absolute top-1 right-1/4 w-4 h-4 bg-red-500 rounded-full border-2 border-[#1a1a24] flex items-center justify-center text-[10px] font-black text-white">
                     {tab.badge}
                   </div>
                )}
              </button>
            )
          })}
        </div>

      </div>
    </div>
  );
}
